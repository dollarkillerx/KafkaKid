# Kafka集群手动部署

### zookeeper集群

zoo1:
```
version: '3'

services:
  zoo1:
    image: zookeeper:3.4
    restart: always
    network_mode: host
    hostname: zoo1
    container_name: zoo1
    ports:
      - 2181:2181
      - 2888:2888
      - 3888:3888
    volumes:
      - "./zoo1/data:/data"
      - "./zoo1/datalog:/datalog"
    environment:
      ZOO_MY_ID: 1
      ZOO_SERVERS: server.1=0.0.0.0:2888:3888 server.2=192.168.88.40:2888:3888 server.3=192.168.88.41:2888:3888
```

### kafka集群

kafka1:
``` 
version: '3'

services:
  kafka1:
    image: wurstmeister/kafka
    restart: always
    hostname: kafka1
    container_name: kafka1
    ports:
      - "9092:9092"
    environment:
      KAFKA_ADVERTISED_LISTENERS: PLAINTEXT://192.168.88.39:9092
      KAFKA_ADVERTISED_HOST_NAME: 192.168.88.39
      KAFKA_ADVERTISED_PORT: 9092
      KAFKA_ZOOKEEPER_CONNECT: 192.168.88.39:2181,192.168.88.40:2181,192.168.88.41:2181
      KAFKA_MAX_REQUEST_SIZE: 104857600
      KAFKA_MESSAGE_MAX_BYTES: 115343360
      KAFKA_FETCH_MAX_BYTES: 125829120
      JMX_PORT: 6501
      KAFKA_ACKS: "all"
    volumes:
      - ./kafka1/logs:/kafka    
```

### kowl & kafka-manager监控
kowl_config.yaml
``` 
kafka:
  brokers:
    - 192.168.88.39:9092
    - 192.168.88.40:9092
    - 192.168.88.41:9092
```

``` 
version: '3'

services:
  kowl:
    image: quay.io/cloudhut/kowl:v1.1.0
    restart: on-failure
    container_name: kowl
    hostname: kowl
    volumes:
      - ./configs/kowl_config.yaml:/etc/kowl/config.yaml
    ports:
      - "9009:8080"
    entrypoint: ./kowl --config.filepath=/etc/kowl/config.yaml

  kafka-manager:
    image: sheepkiller/kafka-manager:latest
    restart: always
    container_name: kafka-manager
    hostname: kafka-manager
    ports:
      - "9006:9000"
    environment:
      ZK_HOSTS: 192.168.88.39:2181,192.168.88.40:2181,192.168.88.41:2181
      KAFKA_BROKERS: 192.168.88.39:9092,192.168.88.40:9092,192.168.88.41:9092
      APPLICATION_SECRET: letmein
      KM_ARGS: -Djava.net.preferIPv4Stack=true
```
